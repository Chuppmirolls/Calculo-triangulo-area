Calculo-triangulo-area
======================

//Calcular a área do triângulo de forma simples, apenas com númeos inteiros//

#include <stdio.h>
int main ()
{
  int base;
  int altura;
  int area;
  
  printf ("Digite o Valor da Base:");
  scanf ("%i", &base);
  
  printf ("Digite o Valor da Altura");
  scanf ("%i", &altura);
  
  area=base * altura / 2
  
  printf("O valor da area do triangulo e=%i",area);
  
  return 0;
}
